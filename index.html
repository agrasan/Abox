import com.badlogic.gdx.*; import com.badlogic.gdx.graphics.*; import com.badlogic.gdx.graphics.glutils.ShapeRenderer; import com.badlogic.gdx.utils.Array; public class AsteroidDodger extends ApplicationAdapter { private ShapeRenderer shapeRenderer; private float playerX, playerY; // Object Pool to prevent garbage collection stutters on low-end PCs private static class Asteroid { float x, y, radius, speed; boolean active; } private Array pool = new Array<>(); @Override public void create() { shapeRenderer = new ShapeRenderer(); playerX = 400; playerY = 100; for (int i = 0; i < 50; i++) pool.add(new Asteroid()); } @Override public void render() { Gdx.gl.glClearColor(0.05f, 0.05f, 0.07f, 1); Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT); // Input Handling if (Gdx.input.isKeyPressed(Input.Keys.LEFT)) playerX -= 5; if (Gdx.input.isKeyPressed(Input.Keys.RIGHT)) playerX += 5; // Logic Update for (Asteroid a : pool) { if (!a.active && Math.random() < 0.02) { a.x = (float) Math.random() * 800; a.y = 600; a.radius = 10; a.speed = 3; a.active = true; } if (a.active) { a.y -= a.speed; if (a.y < 0) a.active = false; } } // Draw shapeRenderer.begin(ShapeRenderer.ShapeType.Filled); shapeRenderer.setColor(Color.CYAN); shapeRenderer.circle(playerX, playerY, 15); // Draw Player shapeRenderer.setColor(Color.GRAY); for (Asteroid a : pool) if (a.active) shapeRenderer.circle(a.x, a.y, a.radius); shapeRenderer.end(); } }
